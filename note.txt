# Zum Docker bauen wichtig !!!!

docker buildx use multi
docker buildx inspect --bootstrap

docker buildx build \
  --platform linux/amd64,linux/arm64 \
  -t finndotde/speedliner:1.0.1 \
  -t finndotde/speedliner:latest \
  --push .


docker compose pull speedliner-server
docker compose up -d --force-recreate speedliner-server
